### Default example

```jsx
import React from 'react';

import { SvgViewer } from '@texttree/svg-viewer-poc';
const marks = { mapid34: 'Mark34', mapid21: 'Mark21', mapid82: 'Mark82' };
<SvgViewer
  link={
    'https://git.door43.org/BSA/ru_biblemap/raw/branch/master/maps/babylonian-empire.svg'
  }
  marks={marks}
/>;
```
